//   + Snapshots completos en appointments (servicio, duración, dirección).
//   + Campos de seguridad en users (verificación, lockout, soft delete).

// ============================================================
//  MÓDULO: USUARIOS Y AUTENTICACIÓN
// ============================================================

Table users {
  id                      uuid        [pk, default: `gen_random_uuid()`]
  email                   varchar(255) [not null, unique, note: "Email único para login"]
  password_hash           varchar(255) [null, note: "null si el user se registró sólo con OAuth"]
  phone                   varchar(30)  [null, note: "Con código de país, ej: +541123456789"]
  role                    user_role    [not null, note: "OWNER | PROFESSIONAL | CLIENT | ADMIN"]
  status                  user_status  [not null, default: "PENDING"]
  avatar_url              text         [null]
  email_verified_at       timestamp    [null, note: "Marca la verificación del email"]
  phone_verified_at       timestamp    [null, note: "Marca la verificación del teléfono"]
  terms_accepted_at       timestamp    [null, note: "Aceptación explícita de T&C — requerido en registro"]
  last_login_at           timestamp    [null]
  failed_login_attempts   integer      [not null, default: 0, note: "Para bloqueo por fuerza bruta"]
  locked_until            timestamp    [null, note: "Cuenta bloqueada hasta esta fecha"]
  deleted_at              timestamp    [null, note: "Soft delete — preservar histórico de transacciones"]
  created_at              timestamp    [not null, default: `now()`]
  updated_at              timestamp    [not null, default: `now()`]

  indexes {
    email
    role
    (status, deleted_at)
  }

  Note: "Tabla base de autenticación. Cada rol tiene su tabla de perfil extendida."
}

Enum user_role {
  OWNER        [note: "Dueño de local/clínica"]
  PROFESSIONAL [note: "Profesional independiente (barbero, odontólogo, etc.)"]
  CLIENT       [note: "Cliente final que reserva turnos"]
  ADMIN        [note: "Administrador de la plataforma"]
}

Enum user_status {
  PENDING    [note: "Recién registrado, pendiente de verificación"]
  ACTIVE     [note: "Habilitado para operar"]
  SUSPENDED  [note: "Suspendido temporalmente"]
  BANNED     [note: "Bloqueado permanentemente"]
}

// Sesiones activas (refresh tokens). Permite logout selectivo por dispositivo.
Table auth_sessions {
  id              uuid        [pk, default: `gen_random_uuid()`]
  user_id         uuid        [not null, ref: > users.id]
  refresh_token   varchar(512) [not null, unique, note: "Hash del refresh token"]
  user_agent      text         [null]
  ip_address      varchar(45)  [null, note: "IPv4 o IPv6"]
  expires_at      timestamp    [not null]
  revoked_at      timestamp    [null]
  created_at      timestamp    [not null, default: `now()`]

  indexes {
    user_id
    expires_at
  }

  Note: "Sesiones activas. Logout = revoked_at. Limpieza periódica de expiradas."
}

// Vinculación con providers OAuth (Google, Apple).
Table social_accounts {
  id              uuid        [pk, default: `gen_random_uuid()`]
  user_id         uuid        [not null, ref: > users.id]
  provider        oauth_provider [not null]
  provider_uid    varchar(255) [not null, note: "ID del usuario en el provider externo"]
  email_at_link   varchar(255) [null, note: "Email recibido del provider al momento del link"]
  created_at      timestamp    [not null, default: `now()`]

  indexes {
    (provider, provider_uid) [unique]
    user_id
  }

  Note: "Login social. La app permite Google y Apple en Auth.tsx."
}

Enum oauth_provider {
  GOOGLE
  APPLE
}

// Tokens de recuperación de contraseña (single-use, expirable).

  expires_at      timestamp    [not null]
  used_at         timestamp    [null]
  created_at      timestamp    [not null, default: `now()`]

  Note: "Soporta el flujo '¿Olvidaste tu contraseña?' del Auth."
}

// Tokens para verificar email/teléfono.
Table verification_tokens {
  id              uuid        [pk, default: `gen_random_uuid()`]
  user_id         uuid        [not null, ref: > users.id]
  channel         verification_channel [not null]
  token_hash      varchar(255) [not null, unique]
  expires_at      timestamp    [not null]
  consumed_at     timestamp    [null]
  created_at      timestamp    [not null, default: `now()`]

  Note: "Emails y SMS de verificación. Un único token activo por canal por user."

Enum verification_channel {

  PHONE

// ============================================================
//  MÓDULO: PROFESIONALES
// ============================================================

Table professional_profiles {
  id                  uuid        [pk, default: `gen_random_uuid()`]
  user_id             uuid        [not null, unique, ref: > users.id]
  first_name          varchar(100) [not null]
  last_name           varchar(100) [not null]
  bio                 text         [null, note: "Descripción pública del profesional"]
  specialty           varchar(100) [not null, note: "Ej: Barbero, Odontólogo, Esteticista"]
  category_id         uuid         [not null, ref: > categories.id, note: "Rubro principal"]
  public_slug         varchar(100) [not null, unique, note: "URL pública: /pro/mi-nombre"]
  is_visible          boolean      [not null, default: true, note: "Toggle público en ProProfile.tsx — false = oculto en búsquedas"]
  years_experience    integer      [null, note: "Para mostrar en perfil público"]
  instagram_url       text         [null]
  website_url         text         [null]

  // Verificación
  validation_status   prof_validation_status [not null, default: "PENDING"]
  validated_at        timestamp    [null]
  validated_by        uuid         [null, ref: > users.id, note: "Admin que validó"]

  // Comisiones (RN-04)
  commission_rate     decimal(5,2) [not null, default: 15.00, note: "% de comisión que cobra la plataforma"]
  monthly_bookings    integer      [not null, default: 0, note: "Contador del mes en curso"]
  commission_tier     commission_tier [not null, default: "STANDARD"]

  // Caches denormalizados (recálculo al insertar review / completar booking)
  average_rating      decimal(3,2) [not null, default: 0, note: "Cache de rating promedio (0–5)"]
  total_reviews       integer      [not null, default: 0]
  total_completed     integer      [not null, default: 0, note: "Histórico total de turnos completados"]

  created_at          timestamp    [not null, default: `now()`]
  updated_at          timestamp    [not null, default: `now()`]

  indexes {
    category_id
    (is_visible, validation_status) [note: "Filtro principal de búsqueda pública"]
    public_slug
  }

  Note: "Perfil extendido del Profesional. Incluye estado de validación, tarifa de comisión y caches de rating."
}

Enum prof_validation_status {
  PENDING    [note: "Documentación enviada, esperando revisión"]
  APPROVED   [note: "Validado por la plataforma, puede operar"]
  REJECTED   [note: "Documentación rechazada"]
  SUSPENDED  [note: "Validación revocada"]
}

Enum commission_tier {
  STANDARD   [note: "Comisión base (ej: 15%)"]
  SILVER     [note: "Comisión reducida por hito (ej: 12%) - RN-04"]
  GOLD       [note: "Comisión mínima por alto volumen (ej: 8%)"]
}

Table transactions {
  id                    uuid        [pk, default: `gen_random_uuid()`]
  appointment_id        uuid        [null, ref: > appointments.id, note: "Si es pago por turno"]
  space_booking_id      uuid        [null, ref: > space_bookings.id, note: "Si es alquiler de espacio"]
  payer_user_id         uuid        [not null, ref: > users.id]
  parent_transaction_id uuid        [null, ref: > transactions.id, note: "Transacción original si esta es un refund"]
  total_amount          decimal(10,2) [not null]
  currency              varchar(3)   [not null, default: "ARS"]
  status                transaction_status [not null, default: "PENDING"]
  payment_method        payment_method [null]
  gateway_reference     varchar(255) [null, note: "ID externo (MercadoPago, Stripe, etc.)"]
  gateway_response      jsonb        [null, note: "Respuesta raw del gateway (encriptada en reposo)"]
  idempotency_key       varchar(255) [null, unique, note: "Previene cobros duplicados ante retries"]
  refund_reason         text         [null]
  processed_at          timestamp    [null]
  refunded_at           timestamp    [null]
  created_at            timestamp    [not null, default: `now()`]
  updated_at            timestamp    [not null, default: `now()`]

  indexes {
    (payer_user_id, created_at)
    appointment_id
    space_booking_id
    status
  }

  Note: "Registro atómico de cada transacción. ACID garantizado (RNF-1)."
}

Enum transaction_status {
  PENDING   [note: "Procesando en el gateway"]
  SUCCESS   [note: "Pago aprobado"]
  FAILED    [note: "Pago rechazado"]
  REFUNDED  [note: "Fondos devueltos"]
}

Enum payment_method {
  CARD
  WALLET
  TRANSFER
  CASH
  MERCADO_PAGO
}

Table payment_splits {
  id                uuid        [pk, default: `gen_random_uuid()`]
  transaction_id    uuid        [not null, ref: > transactions.id]
  recipient_user_id uuid        [not null, ref: > users.id, note: "Destinatario: Propietario, Profesional o Plataforma"]
  role              split_role  [not null]
  amount            decimal(10,2) [not null]
  percentage        decimal(5,2)  [not null]
  status            split_status  [not null, default: "PENDING"]
  liquidated_at     timestamp    [null, note: "Fecha efectiva de acreditación"]
  created_at        timestamp    [not null, default: `now()`]

  indexes {
    transaction_id
    (recipient_user_id, status)
  }

  Note: "Desglose del split automático (RF-4.2): propietario, profesional, plataforma."
}

Enum split_role {
  OWNER        [note: "Tarifa de alquiler del espacio"]
  PROFESSIONAL [note: "Honorarios del profesional"]
  PLATFORM     [note: "Comisión de la plataforma"]
}

Enum split_status {
  PENDING     [note: "Pendiente de acreditación"]
  LIQUIDATED  [note: "Acreditado en la cuenta del destinatario"]
  HELD        [note: "Retenido (ej: disputa, penalidad)"]
}

Table no_show_penalties {
  id                uuid        [pk, default: `gen_random_uuid()`]
  appointment_id    uuid        [not null, ref: > appointments.id]
  client_id         uuid        [not null, ref: > client_profiles.id]
  amount            decimal(10,2) [not null, note: "Monto de la penalidad cobrada (RN-03)"]
  transaction_id    uuid         [null, ref: > transactions.id, note: "Transacción donde se cobró"]
  applied_at        timestamp    [not null, default: `now()`]

  indexes {
    client_id
  }

  Note: "Penalidades aplicadas a clientes que no asisten. Implementa RN-03."
}
